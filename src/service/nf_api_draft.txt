typedef struct    _NF_NOTIFY_INFO 
{ 
   guint      type; 
   GTimeVal   timestamp; 
   union {       
      struct    {   guint       params[4];   } d; 
      struct    {   gchar       chmap[16];   } c; 
      struct    {   guint       len;             
               gpointer    ptr;        
               guint       reserved[2]; } p; 
   }; 
} NF_NOTIFY_INFO; 


1) motion,alarm,sensor,vloss 
   NF_NOTIFY_PARAM type의 NF_NOTIFY_INFO를 사용    
      d.params[0]      아날로그 채널별 비트 마스크 
      d.params[1]   ipcamera 채널별 비트 마스크 
             
2) analog_rec, ipcam_rec 
   NF_NOTIFY_CHMAP 
      c.chmap[0~15]   'T'   timer rec 
            'P'   panic 
            'A' alarm 
            'M' motion 
            ' '   none 
                                           
3) enc_status   - 녹화중인 채널의 갯수 
   NF_NOTIFY_PARAM type의 NF_NOTIFY_INFO를 사용 
      d.params[0]      현재 analog fps 합계 
      d.params[1]      현재 ipcamera fps 합계 
      d.params[2]      max analog fps 
      d.params[3]      max ipcamera fps 

4) net_status   - 네트워크 클라이언트 상태 
   NF_NOTIFY_PARAM type의 NF_NOTIFY_INFO를 사용 
      d.params[0]      접속중인 클라이언트 
      d.params[1]      live mode 클라이언트 
      d.params[2]      playback mode 클라이언트 

5) net_rxtx   - 네트워크 트래픽 모니터 
   NF_NOTIFY_PARAM type의 NF_NOTIFY_INFO를 사용 
      d.params[0]      netsvr status 100~0 
                  트래픽이 원할한지를 표현함 
                  (프레임 레이트 컨트롤 변수로 확인해야겠음) 
      d.params[1]      netsvr rx    
      d.params[2]      netsvr tx    

6) disk_usage   - 디스크 사용량 
   NF_NOTIFY_PARAM type의 NF_NOTIFY_INFO를 사용 
      d.params[0]      사용량 (MB) 
      d.params[1]      총용량 (MB) 


//  notify callback 사용하기 
void
dummy_notify_api( NF_NOTIFY_INFO *pinfo, gpointer data )
{
	g_message("%s data[%s]", __FUNCTION__, data);
	
	g_message("%s type[%d] [%d][%d][%d][%d] ", __FUNCTION__, 
					pinfo->type,
					pinfo->d.params[0],
					pinfo->d.params[1],
					pinfo->d.params[2],
					pinfo->d.params[3] );
	
}
      
{
	gulong 		cb_handle = 0;
	gboolean	cb_remove_ret = 0;
	NF_NOTIFY_INFO	*pinfo;
	
	cb_handle= nf_api_notify_connect_cb( "sensor", dummy_notify_api, "sensor");
	g_message("%s connected cb_handle[%d]",__FUNCTION__, cb_handle);

	cb_handle= nf_api_notify_connect_cb( "sensor", dummy_notify_api, "sensor");
	g_message("%s connected cb_handle[%d]",__FUNCTION__, cb_handle);
			
	pinfo = nf_api_notify_new();
		
	pinfo->d.params[0] = 3;
	pinfo->d.params[1] = 3;
	pinfo->d.params[2] = 9;
	pinfo->d.params[3] = 8;
		
	g_object_set(_nf_notify, "sensor", pinfo, NULL);
	g_message("%s object set",__FUNCTION__);
		
	cb_remove_ret = nf_api_notify_remove_cb ("sensor", cb_handle);
	g_message("%s remove cb_handle ret[%d]",__FUNCTION__, cb_remove_ret);	
	
	pinfo->d.params[3] = 9;
	
	g_object_set(_nf_notify, "sensor", pinfo, NULL);
	g_message("%s object set",__FUNCTION__);
		
	nf_api_notify_free( pinfo);
	
}

{
	guint i = 0;
	gchar buff[128];

/*	
<item key="sys.test.uint"				type="UINT" 	min="" max="" def="" val="33" />
<item key="sys.test.int"				type="INT"	 	min="" max="" def="" val="-2" />
<item key="sys.test.bool"				type="BOOL" 	min="" max="" def="" val="1" />
*/	
	g_message("----------------------------------------");
	//while(1)
	{
		GValue 		ret_value = {0,};
		GValue 		set_value = {0,};

		g_message ( "NF_HOST sys.test.uint [%d]", g_value_get_uint(&ret_value));
		g_value_unset(&ret_value);

		nf_api_sysdb_get_key0("sys.test.int", &ret_value, NULL);
		g_message ( "NF_HOST sys.test.int [%d]", g_value_get_int(&ret_value));
		g_value_unset(&ret_value);
		
		nf_api_sysdb_get_key0("sys.test.bool", &ret_value, NULL);
		g_message ( "NF_HOST sys.test.bool [%d]", g_value_get_boolean(&ret_value));
		g_value_unset(&ret_value);
				
		nf_api_sysdb_get_key0("sys.info.sysid", &ret_value, NULL);
		g_message ( "NF_HOST SYSID [%s]", g_value_get_string(&ret_value));
		g_value_unset(&ret_value);
		
		nf_api_sysdb_get_key0("sys.info.swver", &ret_value, NULL);
		g_message ( "NF_HOST SWVER [%s]", g_value_get_string(&ret_value));
		g_value_unset(&ret_value);
				
		
		g_value_init( &set_value,  G_TYPE_STRING);
		g_snprintf(buff,sizeof(buff), "test software id_%d", ++i);
		g_value_set_string(&set_value, buff);		
		nf_api_sysdb_set_key0("sys.info.swver", &set_value, NULL);						
		g_value_unset(&set_value);
			
		nf_api_sysdb_get_key0("sys.info.swver", &ret_value, NULL);
		g_message ( "NF_HOST SWVER [%s]", g_value_get_string(&ret_value));
		g_value_unset(&ret_value);

		usleep(1);		
	}				
	g_message("----------------------------------------");

#if 0	// testset
